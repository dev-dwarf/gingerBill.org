import (
    "fmt"

// Vector Type

    X float32
    Y float32
    Z float32
}

func dot3(a, b Vector3) float32 {
    return a.X * b.X + a.Y * b.Y + a.Z * b.Z
}

    var a float32 = 1337.0123
    v := Vector3{1, 4, 9}
